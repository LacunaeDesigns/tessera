```js
// Gate 2 — letter-facing copy scans (docs/design-language.md + open-when.md rules).
// Usage: node tools/prose-check.js [files...]   (default: the letter-facing set)
// Exit 0 always (WARNs are for reading, not for CI to hide behind) unless --strict.
'use strict';
const fs = require('fs');
const path = require('path');

const DEFAULT_FILES = ['data/occasions.js', 'index.html', 'js/compose.js',
  'js/print.js', 'js/registry.js', 'js/manifest.js'];

const RULES = [
  { re: /—/g, why: 'em-dash (banned outside interrupted speech)' },
  { re: /\bNot [^.!?\n]{2,40}\. Not /g, why: 'negation-anaphora stack' },
  { re: /\b(hold space|your feelings are valid|self-care|journey of|healing journey)\b/gi, why: 'therapy-speak' },
  { re: /\b(act now|don't miss|hurry|limited time|last chance)\b/gi, why: 'urgency pattern' },
  { re: /\b(delve|tapestry|testament to|in a world|unleash|elevate)\b/gi, why: 'AI-tell vocabulary' },
  { re: /!{2,}/g, why: 'stacked exclamation marks' }
];

let warns = 0;
const files = process.argv.slice(2).filter(a => !a.startsWith('--'));
for (const f of (files.length ? files : DEFAULT_FILES)) {
  const p = path.join(__dirname, '..', f);
  if (!fs.existsSync(p)) { console.log('skip (missing): ' + f); continue; }
  const lines = fs.readFileSync(p, 'utf8').split('\n');
  lines.forEach((line, i) => {
    for (const r of RULES) {
      const m = line.match(r.re);
      if (m) { warns++; console.log(`WARN ${f}:${i + 1} ${r.why} -> ${line.trim().slice(0, 90)}`); }
    }
  });
}
console.log(warns ? `\n${warns} WARN(s). Read each one; fix or justify.` : 'prose: clean');
if (warns && process.argv.includes('--strict')) process.exit(1);
```

- [ ] **Step 2: Run it** — `node tools/prose-check.js`. Expected: WARNs or clean; **read every WARN** and fix or note the justification (code comments that legitimately contain a dash pattern can be ignored with judgment; letter-facing strings cannot).
