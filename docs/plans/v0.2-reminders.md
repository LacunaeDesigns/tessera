> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Remembering without servers: per-letter and whole-registry `.ics` export, plus a printed open-dates card. Spec: [../features/reminders.md](../features/reminders.md).

**Architecture:** `js/ics.js` is a pure, deterministic, fixture-tested RFC 5545 builder (dual-env). Wiring: registry rows + the sealed screen get an "add to calendar" action; `print.js` gains the card as a register-page variant.

**Determinism note:** RFC 5545 requires `DTSTAMP`; `Date.now` is banned in pure modules. The caller passes the letter's `written` date as the stamp — deterministic, testable, and honest (the event was authored when the letter was).

```js
/* ics.js — PURE RFC 5545 builder. No Date.now, no locale, no DOM.
   buildIcs(letters) -> string. letters: [{ id, to, written, openOn, coverText, alarm? }] */
(function () {
  'use strict';
  function esc(s) { return String(s).replace(/\\/g, '\\\\').replace(/;/g, '\\;').replace(/,/g, '\\,').replace(/\n/g, '\\n'); }
  function dateBasic(iso) { return iso.replace(/-/g, ''); }
  function fold(line) { /* RFC 5545 §3.1: 75 octets, continuation lines start with one space */
    var out = [], bytes = 0, cur = '';
    for (var i = 0; i < line.length; i++) {
      var ch = line[i], w = encodeURIComponent(ch).replace(/%[0-9A-F]{2}/g, 'x').length; /* octet count per char */
      if (bytes + w > (out.length ? 74 : 75)) { out.push(cur); cur = ' '; bytes = 1; }
      cur += ch; bytes += w;
    }
    out.push(cur);
    return out.join('\r\n');
  }
  function vevent(l) {
    var lines = [
      'BEGIN:VEVENT',
      'UID:' + l.id + '@tessera',
      'DTSTAMP:' + dateBasic(l.written) + 'T000000Z',
      'DTSTART;VALUE=DATE:' + dateBasic(l.openOn),
      'SUMMARY:' + esc('A letter is ready: ' + l.to + ' (' + l.id + ')'),
      'DESCRIPTION:' + esc(l.coverText)
    ];
    if (l.alarm) lines.push('BEGIN:VALARM', 'ACTION:DISPLAY', 'DESCRIPTION:' + esc('A letter opens today.'), 'TRIGGER:PT9H', 'END:VALARM');
    lines.push('END:VEVENT');
    return lines;
  }
  function buildIcs(letters) {
    var lines = ['BEGIN:VCALENDAR', 'VERSION:2.0', 'PRODID:-//Tessera//letters//EN', 'CALSCALE:GREGORIAN'];
    letters.forEach(function (l) { lines = lines.concat(vevent(l)); });
    lines.push('END:VCALENDAR');
    return lines.map(fold).join('\r\n') + '\r\n';
  }
  var api = { buildIcs: buildIcs };
  if (typeof module !== 'undefined' && module.exports) module.exports = api;
  else this.TesseraIcs = api;
}).call(this);
```
